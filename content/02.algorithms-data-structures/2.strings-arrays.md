### Merge sorted arrays

```ts
function merge(nums1: number[], m: number, nums2: number[], n: number): void {
  if (m + n === m) {
    return;
  }

  let j = 0;

  for (let i = 0; i < nums1.length; i++) {
    const el1 = nums1[i];

    if (j >= nums2.length) break;

    if (el1 === 0) {
      nums1[i] = nums2[j];

      j++;
    }
  }

  nums1.sort((a, b) => a - b);
}
```

### Remove element

```ts
function removeElement(nums: number[], val: number): number {
  let removedItemsCount = 0;

  for (let i = 0; i < nums.length; i++) {
    const el = nums[i];

    if (el === val) {
      nums[i] = 0;

      removedItemsCount++;
    }
  }

  nums.sort((a, b) => b - a);

  const arrLen = nums.length - removedItemsCount;

  return arrLen;
}
```

### Remove duplicates from sorted array

```ts
function removeDuplicates(nums: number[]): number {
  if (nums.length === 0) return 0;
  if (nums.length === 1) return 1;

  let k = 1;

  for (let i = 1; i < nums.length; i++) {
    if (nums[i - 1] !== nums[i]) {
      nums[k] = nums[i];

      k++;
    }
  }

  nums.length = k;

  return k;
}
```

### Majority element

```ts
function majorityElement(nums: number[]): number {
  let obj = {};
  let maxCount = 0;
  let maxNum: number | null = null;

  for (let i = 0; i < nums.length; i++) {
    const el = nums[i];

    obj[el] = (obj[el] || 0) + 1;

    if (obj[el] > maxCount) {
      maxCount = obj[el];

      maxNum = el;
    }
  }

  return maxNum;
}
```

### Rotate array

```ts
function rotate(nums: number[], k: number): void {
  const copy = [...nums];
  const n = nums.length;

  for (let i = 0; i < copy.length; i++) {
    const el = copy[i];

    let j = (i + k) % n;

    nums[j] = el;
  }
}
```

### Best time to buy and sell stock

```ts
function maxProfit(prices: number[]): number {
  let profit = 0;
  let min = prices[0];

  for (let i = 1; i < prices.length; i++) {
    if (prices[i] < min) {
      min = prices[i];
    } else {
      const tempProfit = prices[i] - min;

      if (tempProfit > profit) {
        profit = tempProfit;
      }
    }
  }

  return profit;
}
```

### Jump game

```ts
function canJump(nums: number[]): boolean {
  const n = nums.length;

  let max = 0;

  for (let i = 0; i < n; i++) {
    if (i > max) return false;

    max = Math.max(max, i + nums[i]);

    if (max >= nums.length - 1) return true;
  }

  return true;
}
```

### Jump game II

```ts
function jump(nums: number[]): number {
  const n = nums.length;

  let jumps = 0;
  let counter = 0;
  let biggestJump = 0;

  for (let i = 0; i < n - 1; i++) {
    const jumpAvailable = nums[i];

    biggestJump = Math.max(biggestJump, i + jumpAvailable);

    if (counter === i) {
      jumps++;

      counter = biggestJump;
    }
  }

  return jumps;
}
```

### Insert Delete GetRandom O(1)

```ts
class RandomizedSet {
  private arr: number[];
  private map: Map<number, number>;

  constructor() {
    this.arr = [];
    this.map = new Map();
  }

  insert(val: number): boolean {
    if (this.map.has(val)) return false;

    this.arr.push(val);
    this.map.set(val, this.arr.length - 1);

    return true;
  }

  remove(val: number): boolean {
    if (!this.map.has(val)) return false;

    const index = this.map.get(val)!;
    const last = this.arr[this.arr.length - 1];

    this.arr[index] = last;
    this.map.set(last, index);

    this.arr.pop();
    this.map.delete(val);

    return true;
  }

  getRandom(): number {
    const index = Math.floor(Math.random() * this.arr.length);
    return this.arr[index];
  }
}
```

### Product of array except self

```ts
function productExceptSelf(nums: number[]): number[] {
  const n = nums.length;

  let res: number[] = [];

  let prefix = 1;
  for (let i = 0; i < n; i++) {
    res[i] = prefix;

    prefix *= nums[i];
  }

  let postfix = 1;
  for (let i = n - 1; i >= 0; i--) {
    res[i] *= postfix;

    postfix *= nums[i];
  }

  return res;
}
```

### Gas station

```ts
function canCompleteCircuit(gas: number[], cost: number[]): number {
  let tankFueled = 0;
  let totalTank = 0;

  const totalGas = gas.reduce((acc, val) => (acc += val), 0);
  const totalCost = cost.reduce((acc, val) => (acc += val), 0);

  if (totalGas < totalCost) return -1;

  const n = gas.length;

  let start = 0;

  for (let i = 0; i < n; i++) {
    const diff = gas[i] - cost[i];

    totalTank += diff;
    tankFueled += diff;

    if (tankFueled < 0) {
      start = i + 1;
      tankFueled = null;
    }
  }

  return totalTank >= 0 ? start : -1;
}
```

### Candy

```ts
function candy(ratings: number[]): number {
  const n = ratings.length;

  const candies = new Array(n).fill(1);

  for (let i = 1; i < n; i++) {
    const el = ratings[i];

    const left = ratings[i - 1];

    if (el > left) {
      candies[i] = candies[i - 1] + 1;
    }
  }

  for (let i = n - 2; i >= 0; i--) {
    const el = ratings[i];

    const right = ratings[i + 1];

    if (el > right && candies[i] <= candies[i + 1]) {
      candies[i] = candies[i + 1] + 1;
    }
  }

  return candies.reduce((acc, val) => acc + val, 0);
}
```

```ts
function trap(height: number[]): number {
  const n = height.length;

  if (n <= 2) return 0;

  let left = 0;
  let right = n - 1;

  let leftHigh = 0;
  let rightHigh = 0;

  let water = 0;

  while (left < right) {
    if (height[left] < height[right]) {
      if (height[left] > leftHigh) {
        leftHigh = height[left];
      } else {
        water += leftHigh - height[left];
      }

      left++;
    } else {
      if (height[right] > rightHigh) {
        rightHigh = height[right];
      } else {
        water += rightHigh - height[right];
      }

      right--;
    }
  }

  return water;
}
```

### Roman to integer

```ts
function romanToInt(s: string): number {
  const SYMBOLS = {
    I: 1,
    V: 5,
    X: 10,
    L: 50,
    C: 100,
    D: 500,
    M: 1000,
  };

  const SPECIALS = {
    IV: 4,
    IX: 9,
    XL: 40,
    XC: 90,
    CD: 400,
    CM: 900,
  };

  let result = 0;

  const arr = s.split("");

  for (let i = 0; i < arr.length; i++) {
    const el = arr[i];

    if (i !== arr.length - 1) {
      const nextEl = arr[i + 1];
      const doubleEl = el + nextEl;

      if (SPECIALS[doubleEl] !== undefined) {
        result += SPECIALS[doubleEl];
        i++;
        continue;
      }
    }

    result += SYMBOLS[el];
  }

  return result;
}
```

### Integer to roman

```ts
function intToRoman(num: number): string {
  const VALUES = [1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1];

  const SYMBOLS = [
    "M",
    "CM",
    "D",
    "CD",
    "C",
    "XC",
    "L",
    "XL",
    "X",
    "IX",
    "V",
    "IV",
    "I",
  ];

  let res = "";
  let val = num;

  for (let i = 0; i < VALUES.length; i++) {
    while (val >= VALUES[i]) {
      val -= VALUES[i];
      res += SYMBOLS[i];
    }
  }

  return res;
}
```

### Length of last word

```ts
function lengthOfLastWord(s: string): number {
  const rev = s.split("").reverse().join("").trim();

  for (let i = 0; i < rev.length; i++) {
    const ch = rev[i];

    if (ch === " ") return i;
  }

  return rev.length;
}
```

### Longest common prefix

```ts
function longestCommonPrefix(strs: string[]): string {
  const arr = strs.sort();

  const first = arr[0];
  const last = arr[arr.length - 1];

  let i = 0;
  while (i < first.length && first[i] === last[i]) {
    i++;
  }

  return first.slice(0, i) || "";
}
```

### Zigzag conversion

```ts
function convert(s: string, numRows: number): string {
  if (numRows === 1 || s.length <= numRows) return s;

  const step = 2 * numRows - 2;

  let res = "";

  for (let row = 0; row < numRows; row++) {
    for (let i = row; i < s.length; i += step) {
      res += s[i];

      const j = i + step - 2 * row;

      if (row != 0 && row != numRows - 1 && j < s.length) {
        res += s[j];
      }
    }
  }

  return res;
}
```

### Text justification

```ts
function fullJustify(words: string[], maxWidth: number): string[] {
  const wordsLen = words.length;
  const totalLen = words.reduce((acc, str) => (acc += str.length), 0);

  let res = [];
  let currLen = 0;
  let currWords: string[] = [];

  for (let i = 0; i < wordsLen; i++) {
    const word = words[i];

    if (currLen + word.length + currWords.length > maxWidth) {
      const gaps = currWords.length - 1;
      const spaces = maxWidth - currLen;

      // one word
      if (gaps === 0) {
        res.push(currWords[0] + " ".repeat(spaces));
      } else {
        const spacePerGap = Math.floor(spaces / gaps);
        let extra = spaces % gaps;

        let line = "";

        for (let i = 0; i < currWords.length; i++) {
          line += currWords[i];

          if (i < gaps) {
            line += " ".repeat(spacePerGap + (extra > 0 ? 1 : 0));
            if (extra > 0) extra--;
          }
        }

        res.push(line);
      }

      currWords = [];
      currLen = 0;
    }

    currWords.push(word);
    currLen += word.length;
  }

  const lastLine = currWords.join(" ");
  res.push(lastLine + " ".repeat(maxWidth - lastLine.length));

  return res;
}
```
